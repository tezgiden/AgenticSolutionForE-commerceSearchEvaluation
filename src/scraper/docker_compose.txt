# Enterprise Web Scraper - Docker Compose Configuration
version: '3.8'

services:
  # Main scraper application
  scraper:
    build: .
    container_name: enterprise-scraper
    restart: unless-stopped
    environment:
      - PYTHONUNBUFFERED=1
      - LOG_LEVEL=INFO
      - ENVIRONMENT=production
      - DATABASE_URL=postgresql://scraper:password@postgres:5432/scraper_db
      - REDIS_URL=redis://redis:6379/0
      - METRICS_ENABLED=true
    volumes:
      - ./config:/app/config:ro
      - ./data:/app/data
      - ./logs:/app/logs
      - ./reports:/app/reports
      - /dev/shm:/dev/shm  # Shared memory for Chrome
    depends_on:
      - postgres
      - redis
    networks:
      - scraper-network
    # For CLI usage
    command: ["tail", "-f", "/dev/null"]
    
  # Web API service
  api:
    build: .
    container_name: scraper-api
    restart: unless-stopped
    ports:
      - "8000:8000"
    environment:
      - PYTHONUNBUFFERED=1
      - LOG_LEVEL=INFO
      - ENVIRONMENT=production
      - DATABASE_URL=postgresql://scraper:password@postgres:5432/scraper_db
      - REDIS_URL=redis://redis:6379/0
      - API_HOST=0.0.0.0
      - API_PORT=8000
    volumes:
      - ./config:/app/config:ro
      - ./data:/app/data
      - ./logs:/app/logs
      - ./reports:/app/reports
    depends_on:
      - postgres
      - redis
    networks:
      - scraper-network
    # Uncomment when API module is implemented
    # command: ["python", "-m", "api_server"]
    command: ["python", "-c", "print('API service placeholder - implement api_server.py')"]
    
  # Background worker for async tasks
  worker:
    build: .
    container_name: scraper-worker
    restart: unless-stopped
    environment:
      - PYTHONUNBUFFERED=1
      - LOG_LEVEL=INFO
      - ENVIRONMENT=production
      - DATABASE_URL=postgresql://scraper:password@postgres:5432/scraper_db
      - REDIS_URL=redis://redis:6379/0
      - WORKER_CONCURRENCY=4
    volumes:
      - ./config:/app/config:ro
      - ./data:/app/data
      - ./logs:/app/logs
      - ./reports:/app/reports
      - /dev/shm:/dev/shm  # Shared memory for Chrome
    depends_on:
      - postgres
      - redis
    networks:
      - scraper-network
    # Uncomment when worker module is implemented
    # command: ["python", "-m", "worker"]
    command: ["python", "-c", "print('Worker service placeholder - implement worker.py')"]
    
  # Scheduler for periodic tasks
  scheduler:
    build: .
    container_name: scraper-scheduler
    restart: unless-stopped
    environment:
      - PYTHONUNBUFFERED=1
      - LOG_LEVEL=INFO
      - ENVIRONMENT=production
      - DATABASE_URL=postgresql://scraper:password@postgres:5432/scraper_db
      - REDIS_URL=redis://redis:6379/0
    volumes:
      - ./config:/app/config:ro
      - ./data:/app/data
      - ./logs:/app/logs
      - ./reports:/app/reports
    depends_on:
      - postgres
      - redis
    networks:
      - scraper-network
    # Uncomment when scheduler module is implemented
    # command: ["python", "-m", "scheduler"]
    command: ["python", "-c", "print('Scheduler service placeholder - implement scheduler.py')"]

  # PostgreSQL database
  postgres:
    image: postgres:15-alpine
    container_name: scraper-postgres
    restart: unless-stopped
    environment:
      - POSTGRES_DB=scraper_db
      - POSTGRES_USER=scraper
      - POSTGRES_PASSWORD=password
      - POSTGRES_INITDB_ARGS=--encoding=UTF-8 --lc-collate=C --lc-ctype=C
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init_db.sql:/docker-entrypoint-initdb.d/init_db.sql:ro
    ports:
      - "5432:5432"
    networks:
      - scraper-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U scraper -d scraper_db"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Redis for caching and task queues
  redis:
    image: redis:7-alpine
    container_name: scraper-redis
    restart: unless-stopped
    command: redis-server --appendonly yes --maxmemory 256mb --maxmemory-policy allkeys-lru
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    networks:
      - scraper-network
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Prometheus for metrics collection
  prometheus:
    image: prom/prometheus:latest
    container_name: scraper-prometheus
    restart: unless-stopped
    ports:
      - "9090:9090"
    volumes:
      - ./monitoring/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus_data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--web.console.libraries=/etc/prometheus/console_libraries'
      - '--web.console.templates=/etc/prometheus/consoles'
      - '--storage.tsdb.retention.time=200h'
      - '--web.enable-lifecycle'
    networks:
      - scraper-network

  # Grafana for visualization
  grafana:
    image: grafana/grafana:latest
    container_name: scraper-grafana
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
      - GF_USERS_ALLOW_SIGN_UP=false
    volumes:
      - grafana_data:/var/lib/grafana
      - ./monitoring/grafana:/etc/grafana/provisioning:ro
    depends_on:
      - prometheus
    networks:
      - scraper-network

  # Nginx reverse proxy
  nginx:
    image: nginx:alpine
    container_name: scraper-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./static:/var/www/static:ro
    depends_on:
      - api
      - grafana
    networks:
      - scraper-network

  # Elasticsearch for log aggregation (optional)
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:8.9.0
    container_name: scraper-elasticsearch
    restart: unless-stopped
    environment:
      - discovery.type=single-node
      - ES_JAVA_OPTS=-Xms512m -Xmx512m
      - xpack.security.enabled=false
    volumes:
      - elasticsearch_data:/usr/share/elasticsearch/data
    ports:
      - "9200:9200"
    networks:
      - scraper-network
    profiles:
      - logging

  # Kibana for log visualization (optional)
  kibana:
    image: docker.elastic.co/kibana/kibana:8.9.0
    container_name: scraper-kibana
    restart: unless-stopped
    environment:
      - ELASTICSEARCH_HOSTS=http://elasticsearch:9200
    ports:
      - "5601:5601"
    depends_on:
      - elasticsearch
    networks:
      - scraper-network
    profiles:
      - logging

  # Logstash for log processing (optional)
  logstash:
    image: docker.elastic.co/logstash/logstash:8.9.0
    container_name: scraper-logstash
    restart: unless-stopped
    volumes:
      - ./monitoring/logstash.conf:/usr/share/logstash/pipeline/logstash.conf:ro
    depends_on:
      - elasticsearch
    networks:
      - scraper-network
    profiles:
      - logging

# Named volumes for data persistence
volumes:
  postgres_data:
    driver: local
  redis_data:
    driver: local
  prometheus_data:
    driver: local
  grafana_data:
    driver: local
  elasticsearch_data:
    driver: local

# Custom network
networks:
  scraper-network:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16
